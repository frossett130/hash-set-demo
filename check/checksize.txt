#include "../lib/hashset.h"

hashset_t *hashset;
hashset_t *testhashset();

#suite The API should be able to check the size and change it.

#tcase Should not change size when the set has been correctly initialised.

#test shouldNotChangeSizeToCorrectlyInitialisedSet
    hashset = testhashset();
    ck_assert_int_eq(0, checksize(hashset));
    freehashset(hashset);

#tcase Should change size when faulty set is provided.

#test shouldChangeSizeToFaultySet
    hashset = testhashset();
    hashset->size = 0;
    ck_assert_int_eq(1, checksize(hashset));
    ck_assert_int_eq(3, hashset->size);
    freehashset(hashset);
