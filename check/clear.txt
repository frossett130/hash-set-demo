#include<stdlib.h>
#include "../lib/hashset.h"

size_t hashcode(char *, size_t);

hashset_t generateHashset() {
    hashset_t retval = init(1);
    struct node * hello = malloc(sizeof(struct node));
    struct node * world = malloc(sizeof(struct node));
    struct node * fun = malloc(sizeof(struct node));
    hello -> value = "hello";
    world -> value = "world";
    fun -> value = "fun";
    hello -> next = world;
    world -> next = fun;
    fun -> next = NULL;
    hello -> prev = NULL;
    world -> prev = hello;
    fun -> prev = world;
    retval.array[0] = hello;
    return retval;
}

#suite clearFunctionShouldPerformExepectedOperation

#test shouldClearGeneratedSet
    hashset_t hashset = generateHashset();
    clear(&hashset);
    ck_assert_int_eq(0, hashset.capacity);
    ck_assert_int_eq(0, hashset.size);
    ck_assert_ptr_null(hashset.array);
