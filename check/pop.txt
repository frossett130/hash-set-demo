#include "../lib/hashset.h"

hashset_t *hashset;

#suite shouldWorkWhenAnElementIsRemovedFromHashSet

#test shouldReturnZeroWhenNonPresentElementRemoved
    ck_assert_int_eq(0, pop(hashset, "tree"));
    ck_assert_int_eq(3, hashset->size);

#test shouldReturnOneWhenPresentElementRemoved
    ck_assert_int_eq(1, pop(hashset, "hello"));
    ck_assert_int_eq(2, hashset->size);
    ck_assert_int_eq(0, ispresent(*hashset, "hello"));

#main-pre
    hashset = init(1);
    struct node *hello = malloc(sizeof(struct node));
    struct node *world = malloc(sizeof(struct node));
    struct node *fun = malloc(sizeof(struct node));
    hello->value = "hello";
    world->value = "world";
    fun->value = "fun";
    hello->next = world;
    world->next = fun;
    fun->next = NULL;
    hello->prev = NULL;
    world->prev = hello;
    fun->prev = world;
    hashset->array[0] = hello;
    hashset->size = 3;

#main-post
    freehashset(hashset);
