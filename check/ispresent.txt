#include<stdlib.h>
#define MOD 128
#include "../lib/hashset.h"

size_t hashcode(char *, size_t);

int nodecontains(hashset_t, char *, struct node **);

hashset_t generateHashset() {
    hashset_t retval = init(1);
    struct node * nodes = calloc(3, sizeof(struct node));
    nodes[0].value = "hello";
    nodes[1].value = "world";
    nodes[2].value = "fun";
    nodes[0].next = nodes + 1;
    nodes[1].prev = nodes;
    nodes[1].next = nodes + 2;
    nodes[2].prev = nodes + 1;
    retval.capacity = 1;
    retval.size = 3;
    retval.array = nodes;
    return retval;
}

#suite makeSureHashCodesAreProducedConsistently

#test shouldCreateMethodsWithConsistentHashCodes
    size_t hashCode0 = hashcode("hello", MOD);
    size_t hashCode1 = hashcode("world", MOD);
    size_t hashCode2 = hashcode("foobar", MOD);
    size_t hashCode3 = hashcode("hello", MOD);

    ck_assert_int_eq(hashCode0, hashCode3);
    ck_assert_int_ne(hashCode0, hashCode1);
    ck_assert_int_ne(hashCode0, hashCode2);
    ck_assert_int_ne(hashCode1, hashCode2);

#test shouldCreateHashNumberLessThanMod
    ck_assert_int_lt(hashcode("hello", MOD), MOD);
    ck_assert_int_lt(hashcode("whatever", MOD), MOD);
    ck_assert_int_lt(hashcode("another string", MOD), MOD);
    ck_assert_int_lt(hashcode("another string again", MOD), MOD);


#suite nodecontainsFunctionShouldWorkAsExpected

#test shouldReturnHelloWhenQueried
    struct node * noderef;
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(1, nodecontains(hashset, "hello", &noderef));
    ck_assert_str_eq("hello", noderef -> value);

#test shouldReturnWorldWhenQueried
    struct node * noderef;
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(1, nodecontains(hashset, "world", &noderef));
    ck_assert_str_eq("world", noderef -> value);

#test shouldReturnFunWhenQueried
    struct node * noderef;
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(1, nodecontains(hashset, "fun", &noderef));
    ck_assert_str_eq("fun", noderef -> value);

#test shouldReturnLastNodeWhenNotPresentWordQueried
    struct node * noderef;
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(0, nodecontains(hashset, "foobar", &noderef));
    ck_assert_str_eq("fun", noderef -> value);

#test shouldReturnEmptyNodeWhenNothingIsInTheArray
    struct node * noderef;
    hashset_t hashset = init(1);
    ck_assert_int_eq(0, nodecontains(hashset, "foobar", &noderef));
    ck_assert_ptr_null(noderef -> value);
