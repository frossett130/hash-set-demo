#include "../lib/hashset.h"
#include <stdlib.h>

hashset_t *testhashset();

hashset_t *hashset;

#suite shouldBeAbleToPushElementsIntoASetAndChangeSizeAccordingly

#test shouldReturnZeroIfElementAlreadyPresent
    hashset = testhashset();
    ck_assert_int_eq(0, push(&hashset, "hello"));
    freehashset(hashset);

#test shouldEditTheSizeOfTheSetAccordingly
    hashset = testhashset();
    ck_assert_int_eq(1, push(&hashset, "another string"));
    ck_assert_int_eq(1, push(&hashset, "a beautiful string"));
    ck_assert_int_eq(5, hashset->size);
    ck_assert_str_eq("hello", ispresent(*hashset, "hello"));
    ck_assert_str_eq("world", ispresent(*hashset, "world"));
    ck_assert_str_eq("fun", ispresent(*hashset, "fun"));
    ck_assert_str_eq("another string", ispresent(*hashset, "another string"));
    ck_assert_str_eq("a beautiful string", ispresent(*hashset, "a beautiful string"));
    ck_assert_ptr_null(ispresent(*hashset, "nothing"));
    ck_assert_int_eq(8, hashset->capacity);
    freehashset(hashset);

#test shouldPushElementsConsisently
    hashset = init(4);
    ck_assert_int_eq(1, push(&hashset, "hello"));
    ck_assert_int_eq(1, push(&hashset, "world"));
    ck_assert_int_eq(2, hashset->size);
    ck_assert_int_eq(0, checksize(hashset));
    ck_assert_str_eq("hello", ispresent(*hashset, "hello"));
    ck_assert_str_eq("world", ispresent(*hashset, "world"));
    freehashset(hashset);
