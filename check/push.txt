#include "../lib/hashset.h"
#include <stdlib.h>

hashset_t generateHashset() {
    hashset_t retval = init(1);
    struct node * nodes = calloc(3, sizeof(struct node));
    nodes[0].value = "hello";
    nodes[1].value = "world";
    nodes[2].value = "fun";
    nodes[0].next = nodes + 1;
    nodes[1].prev = nodes;
    nodes[1].next = nodes + 2;
    nodes[2].prev = nodes + 1;
    retval.capacity = 1;
    retval.size = 3;
    retval.array = nodes;
    return retval;
}

#suite shouldBeAbleToPushElementsIntoASetAndChangeSizeAccordingly

#test shouldReturnZeroIfElementAlreadyPresent
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(0, push(&hashset, "hello"));

#test shouldEditTheSizeOfTheSetAccordingly
    hashset_t hashset = generateHashset();
    ck_assert_int_eq(1, push(&hashset, "another string"));
    ck_assert_int_eq(8, hashset.capacity);
    ck_assert_int_eq(4, hashset.size);
    ck_assert_int_eq(1, ispresent(hashset, "hello"));
    ck_assert_int_eq(1, ispresent(hashset, "world"));
    ck_assert_int_eq(1, ispresent(hashset, "fun"));
    ck_assert_int_eq(1, ispresent(hashset, "another string"));
    ck_assert_int_eq(0, ispresent(hashset, "nothing"));
